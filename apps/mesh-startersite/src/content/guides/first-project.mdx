# Your First Project

Congratulations! You're ready to build your first Cardano project. This guide will walk you through creating a simple web application that will teach you the fundamentals of blockchain development.

## What We'll Build

We'll create a **Cardano Information Dashboard** that:

- Displays real-time Cardano network information
- Shows current ADA price and market data
- Explains blockchain concepts to visitors
- Demonstrates modern web development practices
- Prepares you for more complex blockchain projects

- HTML5 for structure

- Understand web development basics

```javascript
// Utility functions for the dashboard

// Format numbers with commas
export function formatNumber(num) {
  if (num === null || num === undefined) return "N/A";
  return num.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ",");
}

// Format currency
export function formatCurrency(amount, currency = "USD") {
  if (amount === null || amount === undefined) return "N/A";
  return new Intl.NumberFormat("en-US", {
    style: "currency",
    currency: currency,
  }).format(amount);
}

// Format percentage
export function formatPercentage(value) {
  if (value === null || value === undefined) return "N/A";
  return `${value.toFixed(2)}%`;
}

// Debounce function for API calls
export function debounce(func, wait) {
  let timeout;
  return function executedFunction(...args) {
    const later = () => {
      clearTimeout(timeout);
      func(...args);
    };
    clearTimeout(timeout);
    timeout = setTimeout(later, wait);
  };
}

// Show loading state
export function showLoading(elementId) {
  const element = document.getElementById(elementId);
  if (element) {
    element.innerHTML = '<span class="loading">Loading...</span>';
  }
}

// Show error state
export function showError(elementId, message = "Error loading data") {
  const element = document.getElementById(elementId);
  if (element) {
    element.innerHTML = `<span class="error">${message}</span>`;
  }
}

// Add smooth scrolling for navigation links
export function initSmoothScrolling() {
  document.querySelectorAll('a[href^="#"]').forEach(anchor => {
    anchor.addEventListener("click", function (e) {
      e.preventDefault();
      const target = document.querySelector(this.getAttribute("href"));
      if (target) {
        target.scrollIntoView({
          behavior: "smooth",
          block: "start",
        });
      }
    });
  });
}

// Add scroll effects
export function initScrollEffects() {
  const observerOptions = {
    threshold: 0.1,
    rootMargin: "0px 0px -50px 0px",
  };

  const observer = new IntersectionObserver(entries => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        entry.target.classList.add("animate-in");
      }
    });
  }, observerOptions);

  document.querySelectorAll(".overview-card, .network-card").forEach(card => {
    observer.observe(card);
  });
}
```

```javascript
// API functions for fetching Cardano data

// Base URL for Cardano APIs
const CARDANO_API_BASE = "https://api.coingecko.com/api/v3";
const BLOCKFROST_API_BASE = "https://cardano-mainnet.blockfrost.io/api/v0";

// Fetch ADA price from CoinGecko
export async function fetchAdaPrice() {
  try {
    const response = await fetch(
      `${CARDANO_API_BASE}/simple/price?ids=cardano&vs_currencies=usd,eur&include_24hr_change=true`
    );
    const data = await response.json();

    if (data.cardano) {
      return {
        usd: data.cardano.usd,
        eur: data.cardano.eur,
        change24h: data.cardano.usd_24h_change,
      };
    }
    throw new Error("Invalid data format");
  } catch (error) {
    console.error("Error fetching ADA price:", error);
    return null;
  }
}

// Fetch Cardano network statistics
export async function fetchNetworkStats() {
  try {
    // For demo purposes, we'll use mock data
    // In a real application, you'd use Blockfrost API with authentication
    return {
      currentEpoch: 450,
      totalSupply: 45000000000,
      circulatingSupply: 35000000000,
      stakePools: 3200,
    };
  } catch (error) {
    console.error("Error fetching network stats:", error);
    return null;
  }
}

// Fetch real-time network data (requires Blockfrost API key)
export async function fetchRealTimeData(apiKey) {
  if (!apiKey) {
    console.warn("Blockfrost API key required for real-time data");
    return null;
  }

  try {
    const headers = {
      project_id: apiKey,
    };

    const [epochResponse, supplyResponse, poolsResponse] = await Promise.all([
      fetch(`${BLOCKFROST_API_BASE}/epochs/latest`, { headers }),
      fetch(`${BLOCKFROST_API_BASE}/blocks/latest`, { headers }),
      fetch(`${BLOCKFROST_API_BASE}/pools`, { headers }),
    ]);

    const [epochData, supplyData, poolsData] = await Promise.all([
      epochResponse.json(),
      supplyResponse.json(),
      poolsResponse.json(),
    ]);

    return {
      currentEpoch: epochData.epoch,
      totalSupply: 45000000000, // Fixed supply
      circulatingSupply: supplyData.total_supply || 35000000000,
      stakePools: poolsData.length || 3200,
    };
  } catch (error) {
    console.error("Error fetching real-time data:", error);
    return null;
  }
}

// Update dashboard with fetched data
export function updateDashboard(priceData, networkData) {
  // Update price display
  if (priceData) {
    const priceElement = document.getElementById("ada-price");
    if (priceElement) {
      priceElement.innerHTML = `
                <span class="price-main">$${priceData.usd.toFixed(2)}</span>
                <span class="price-change ${priceData.change24h >= 0 ? "positive" : "negative"}">
                    ${priceData.change24h >= 0 ? "+" : ""}${priceData.change24h.toFixed(2)}%
                </span>
            `;
    }
  }

  // Update network statistics
  if (networkData) {
    const elements = {
      "current-epoch": networkData.currentEpoch,
      "total-supply": formatNumber(networkData.totalSupply),
      "circulating-supply": formatNumber(networkData.circulatingSupply),
      "stake-pools": formatNumber(networkData.stakePools),
    };

    Object.entries(elements).forEach(([id, value]) => {
      const element = document.getElementById(id);
      if (element) {
        element.textContent = value;
      }
    });
  }
}
```

```javascript
      this.initUI();

      // Load initial data
      await this.loadData();

      // Set up auto-refresh
      this.startAutoRefresh();

      // Initialize scroll effects
      initScrollEffects();

      this.isInitialized = true;
      console.log("Dashboard initialized successfully");
    } catch (error) {
      console.error("Failed to initialize dashboard:", error);
      this.showErrorMessage(
        "Failed to load dashboard. Please refresh the page."
      );
    }
  }

  initUI() {
    // Initialize smooth scrolling
    initSmoothScrolling();

    // Add loading states
    this.addLoadingStates();

    // Add error handling
    this.addErrorHandling();

    // Add mobile menu toggle
    this.addMobileMenu();
  }

  addLoadingStates() {
    const loadingElements = [
      "ada-price",
      "current-epoch",
      "total-supply",
      "circulating-supply",
      "stake-pools",
    ];

    loadingElements.forEach(id => {
      const element = document.getElementById(id);
      if (element) {
        element.innerHTML = '<span class="loading">Loading...</span>';
      }
    });
  }

  addErrorHandling() {
    // Global error handler
    window.addEventListener("error", event => {
      console.error("Global error:", event.error);
      this.showErrorMessage(
        "An unexpected error occurred. Please refresh the page."
      );
    });

    // Unhandled promise rejection handler
    window.addEventListener("unhandledrejection", event => {
      console.error("Unhandled promise rejection:", event.reason);
      this.showErrorMessage(
        "Failed to load some data. Please check your connection."
      );
    });
  }

  addMobileMenu() {
    // Add mobile menu toggle button
    const nav = document.querySelector(".nav");
    const mobileMenuButton = document.createElement("button");
    mobileMenuButton.className = "mobile-menu-toggle";
    mobileMenuButton.innerHTML = "‚ò∞";
    mobileMenuButton.setAttribute("aria-label", "Toggle mobile menu");

    mobileMenuButton.addEventListener("click", () => {
      const navMenu = document.querySelector(".nav-menu");
      navMenu.classList.toggle("show");
    });

    // Only show on mobile
    if (window.innerWidth <= 768) {
      nav.appendChild(mobileMenuButton);
    }

    // Handle window resize
    window.addEventListener("resize", () => {
      if (window.innerWidth <= 768 && !nav.contains(mobileMenuButton)) {
        nav.appendChild(mobileMenuButton);
      } else if (window.innerWidth > 768 && nav.contains(mobileMenuButton)) {
        nav.removeChild(mobileMenuButton);
      }
    });
  }

  async loadData() {
    try {
      // Fetch data in parallel
      const [priceData, networkData] = await Promise.all([
        fetchAdaPrice(),
        fetchNetworkStats(),
      ]);

      // Update the dashboard
      updateDashboard(priceData, networkData);

      // Add success indicators
      this.addSuccessIndicators();
    } catch (error) {
      console.error("Error loading data:", error);
      this.showErrorMessage(
        "Failed to load data. Please check your connection."
      );
    }
  }

  addSuccessIndicators() {
    // Add visual feedback for loaded data
    const elements = document.querySelectorAll(".network-card, .overview-card");
    elements.forEach((element, index) => {
      setTimeout(() => {
        element.classList.add("loaded");
      }, index * 100);
    });
  }

  startAutoRefresh() {
    // Refresh data every 5 minutes
    this.updateInterval = setInterval(
      async () => {
        await this.loadData();
      },
      5 * 60 * 1000
    );
  }

  showErrorMessage(message) {
    // Create error banner
    const errorBanner = document.createElement("div");
    errorBanner.className = "error-banner";
    errorBanner.innerHTML = `
            <div class="error-content">
                <span class="error-icon">‚ö†Ô∏è</span>
                <span class="error-message">${message}</span>
                <button class="error-close" onclick="this.parentElement.parentElement.remove()">√ó</button>
            </div>
        `;

    // Add to page
    document.body.insertBefore(errorBanner, document.body.firstChild);

    // Auto-remove after 10 seconds
    setTimeout(() => {
      if (errorBanner.parentElement) {
        errorBanner.remove();
      }
    }, 10000);
  }

  destroy() {
    if (this.updateInterval) {
      clearInterval(this.updateInterval);
    }
  }
}

// Initialize dashboard when DOM is loaded
document.addEventListener("DOMContentLoaded", () => {
  window.dashboard = new CardanoDashboard();
});

// Clean up on page unload
window.addEventListener("beforeunload", () => {
  if (window.dashboard) {
    window.dashboard.destroy();
  }
});
```
